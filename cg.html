4.1 DDA 

#include< graphics.h>
#include< stdio.h>
#include< conio.h>
#include< math.h>
#include< dos.h>
void main()
{
    float x,y,dx,dy,step,x1,y1,x2,y2;
    int i,gd=DETECT,gm;
    initgraph(&gd,&gm,"C:\\TURBOC3\\BGI");
    printf("Enter the value of x1 and y1: ");
    scanf("%f%f",&x1,&y1);
    printf("Enter the value of x2 and y2: ");
    scanf("%f%f",&x2,&y2);
    dy=abs(y2-y1);
    dx=abs(x2-x1);
    if(dx>=dy)
    {
    step=dx;
    }
    else
    {
    step=dy;
    }
    dx=dx/step;
    dy=dy/step;
    x=x1;
    y=y1;
    i=1;
    while(i<=step)
    {
    putpixel(x,y,5);
    x=x+dx;
    y=y+dy;
    i=i+1;
    delay(50);
    }
    getch();
    closegraph();
}



4.2 Bresenhams line drawing algo

#include< stdio.h>
#include< conio.h>
#include< graphics.h>
void drawline(int x1,int y1,int x2,int y2)
{
    int x,y,p,dx,dy;
    dx=x2-x1;
    dy=y2-y1;
    x=x1;
    y=y1;
    p=2*dy-dy;
    while(x=0)
    {
    putpixel(x,y,7);
    y=y+1;
    p=p+2*dy-2*dx;
    }
    else
    {
    putpixel(x,y,7);
    p=p+2*dy;
    }
    x=x+1;
    }
}
void main()
{
    int gd=DETECT,gm,error,x1,x2,y1,y2;
    initgraph(&gd,&gm,"C:\\TURBOC3\\BGI");
    printf("Enter the value of x1 and y1: ");
    scanf("%d%d",&x1,&y1);
    printf("Enter the value of x2 and y2: ");
    scanf("%d%d",&x2,&y2);
    drawline(100,200,300,400);
    getch();
    closegraph();
}


5.1 midpoint circle

#include< stdio.h>
#include< conio.h>
#include< dos.h>
#include< graphics.h>
void drawcircle(int x1,int y1,int r)
{
    int x=r;
    int y=0;
    int p;
    while(x>=y)
    {
    putpixel(x1+x,y1+y,7);
    putpixel(x1+y,y1+x,7);
    putpixel(x1-y,y1+x,7);
    putpixel(x1-x,y1+y,7);
    putpixel(x1-x,y1-y,7);
    putpixel(x1-y,y1-x,7);
    putpixel(x1+y,y1-x,7);
    putpixel(x1+x,y1-y,7);
    if(p<=0)
    {
    y=y+1;
    p=p+2*y+1;
    }
    else
    {
    x=x-1;
    p=p-2*x+1;
    }
    delay(50);
    }
}
void main()
{
    int gd=DETECT,gm,x1,y1,r;
    initgraph(&gd,&gm,"C:\\TURBOC3\\BGI");
    printf("Enter the value of r ");
    scanf("%d",&r);
    printf("Enter the value of x2 and y2: ");
    scanf("%d%d",&x1,&y1);
    drawcircle(x1,y1,r);
    getch();
    closegraph();
}

7.1 2d rotation

#include< stdio.h>
#include< conio.h>
#include< graphics.h>
#include< stdlib.h>
#include< math.h>
void main()
{
    int gd=DETECT,gm,error,x1,x2,y1,y2,xn,yn;
    double r11,r12,r21,r22,th;
    initgraph(&gd,&gm,"C:\\TURBOC3\\BGI");
    printf("Enter the 2 line end points : ");
    scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
    line(x1,y1,x2,y2);
    printf("Enter the value of angle : ");
    scanf("%d",&th);
    r11=cos(th*3.14/180);
    r12=sin(th*3.14/180);
    r21=-sin(th*3.14/180);
    r22=cos(th*3.14/180);
    xn=((r11*x2)-(r12*y2));
    yn=((r21*x2)-(r22*y2));
    line(x1,y1,xn,yn);
    getch();
    closegraph();
}

9.1 floodfill algo

#include< conio.h>
#include< dos.h>
#include< graphics.h>
#include< stdio.h>
void floodFill(int x,int y,int oc,int nc)
{
    if(getpixel(x,y)==oc )
    {
    delay(20);
    putpixel(x,y,nc);
    floodFill(x+1,y,oc,nc);
    floodFill(x,y+1,oc,nc);
    floodFill(x-1,y,oc,nc);
    floodFill(x,y-1,oc,nc);
    }
}
void main()
{
    int gm,gd=DETECT,r;
    int x,y;
    printf("enter x and y position of circle: ");

    printf("enter r: ");
    scanf("%d",&r);
    initgraph(&gd,&gm,"C:\\TURBOC3\\BGI");
    circle(x,y,r);
    floodFill(x,y,0,15);
    delay(50);
    getch();
    closegraph();
}



9.2 boundfill algo

#include< stdio.h>
#include< conio.h>
#include< dos.h>
#include< graphics.h>
#include< stdio.h>
void boundaryfill(int x,int y,int fc,int bc)
{
    if(getpixel(x,y)!=bc && getpixel(x,y)!=fc )
    {
    putpixel(x,y,fc);
    boundaryfill(x+1,y,fc,bc);
    boundaryfill(x,y+1,fc,bc);
    boundaryfill(x-1,y,fc,bc);
    boundaryfill(x,y-1,fc,bc);
    }
}
int main()
{
    int gm,gd=DETECT,r;
    int x,y;
    printf("enter x and y positio for circle: ");
    scanf("%d%d",&x,&y);
    printf("enter r: ");
    scanf("%d",&r);
    initgraph(&gd,&gm,"C:\\TURBOC3\\BGI");
    circle(x,y,r);
    boundaryfill(x,y,4,15);
    delay(50);
    getch();
    closegraph();
    return 0;
}
